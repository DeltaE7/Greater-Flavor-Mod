country_event = { # Checks countries before firing renaming key list event
    id = 71701
    title = "Dynamic Loc Regions Bounce Event"
    #desc = ""
    picture = "commonwealth"
	allow_multiple_instances = no
    is_triggered_only = yes
	
	immediate = {
		# Will only itterate through the list n times where n is countries scoped (to increase performance). Tracking what countries are in what slot should be done in event logic.
		# Immediates are required as to skip the event queue.
		UTI = { set_variable = { which = itterator value = 0 } }
		any_land_province = {
			limit = {
				has_province_modifier = slot_1_region
			}
			UTI = { change_variable = { which = itterator value = 1 } }
		}	
		any_land_province = {
			limit = {
				has_province_modifier = slot_2_region
			}
			UTI = { change_variable = { which = itterator value = 1 } }
		}	
		country_event = 71702
	}
	
	option = {
        name = ""
    }
}


country_event = { # Itterate through key list (Generated in python)
    id = 71702
    title = "Dynamic Loc Regions Rename Event"
    #desc = ""
    picture = "commonwealth"
	allow_multiple_instances = no
    is_triggered_only = yes
	
	immediate = {
		# Immediates are required as to skip the event queue.
		any_land_province = {
			limit = {
				AND = {
					has_province_modifier = slot_2_region
					UTI = { check_variable = { which = itterator value = 1.99 } }
					NOT = { UTI = { check_variable = { which = itterator value = 2.99 } } }
				}
			}
			add_province_modifier = { name = regions_loc_land duration = -1 }
			slot_2 = { add_province_modifier = { name = regions_loc_sea duration = -1 } }
			remove_province_modifier = slot_2_region
		}
		any_land_province = {
			limit = {
				AND = {
					has_province_modifier = slot_1_region
					UTI = { check_variable = { which = itterator value = 0.99 } }
					NOT = { UTI = { check_variable = { which = itterator value = 1.99 } } }
				}
			}
			add_province_modifier = { name = regions_loc_land duration = -1 }
			slot_1 = { add_province_modifier = { name = regions_loc_sea duration = -1 } }
			remove_province_modifier = slot_1_region
		}
		any_sea_province = {
			limit = {
				AND = {
					has_province_modifier = regions_loc_sea
					USA_1 = { has_province_modifier = regions_loc_land }
				}
			}
			secede_province = THIS
			state_scope = { change_region_name = "§bColumbia§!" }
			secede_province = QQQ
		}
		any_sea_province = {
			limit = {
				AND = {
					has_province_modifier = regions_loc_sea
					CAN_60 = { has_province_modifier = regions_loc_land }
				}
			}
			secede_province = THIS
			state_scope = { change_region_name = "§bOther Text§!" }
			secede_province = QQQ
		}
		any_land_province = {
			remove_province_modifier = regions_loc_land
		}
		any_sea_province = {
			remove_province_modifier = regions_loc_sea
		}
		any_country = {
			limit = {
				tag = UTI
				check_variable = { which = itterator value = 0 }
			}
			change_variable = { which = itterator value = -1 }
			CLN = { country_event = 71702 }
		}
	}
	
	option = {
        name = "Change the Name"
    }
}
