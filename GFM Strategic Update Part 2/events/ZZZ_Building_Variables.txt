			}

				}
				secede_province = THIS
				
				state_scope = { change_region_name = "§R1§!" }
				secede_province = QQQ
			}
		}
		random_country = {
			limit = {
				NOT = { check_variable = { which = variable_loc_value value = 2.99 } }
				check_variable = { which = variable_loc_value value = 1.99 }
				has_country_flag = rename_loc_var
			}
			any_sea_province = {
				limit = {
					has_province_modifier = var_rename_modifier
