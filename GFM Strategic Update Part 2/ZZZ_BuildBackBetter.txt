country_event = { # Dynamic Renaming Function for Country Localization
    id = 71602
    title = "TEST"
    desc = "TEST"
    picture = "commonwealth"
	allow_multiple_instances = no
    is_triggered_only = yes
	
	immediate = {
		any_country = {
			limit = {
				NOT = { check_variable = { which = country_loc_slot value = 2.99 } }
				check_variable = { which = country_loc_slot value = 1.99 }
				UTI = { check_variable = { which = itterator value = 1.99 } }
				NOT = { UTI = { check_variable = { which = itterator value = 2.99 } } }
			}
			set_country_flag = rename_loc_region
			slot_2 = { add_province_modifier = { name = loc_rename_modifier duration = -1 } }
		}
		any_country = {
			limit = {
				NOT = { check_variable = { which = country_loc_slot value = 1.99 } }
				check_variable = { which = country_loc_slot value = 0.99 }
				UTI = { check_variable = { which = itterator value = 0.99 } }
				NOT = { UTI = { check_variable = { which = itterator value = 1.99 } } }
			}
			set_country_flag = rename_loc_region
			slot_1 = { add_province_modifier = { name = loc_rename_modifier duration = -1 } }
		}
		random_country = {
			limit = {
				has_country_flag = rename_loc_region
				tag = ENG
			}
			any_sea_province = {
				limit = {
					has_province_modifier = loc_rename_modifier
				}
				secede_province = THIS
				
				state_scope = { change_region_name = "Renamer 1" }
				secede_province = NZL
			}
		}
		random_country = {
			limit = {
				has_country_flag = rename_loc_region
				tag = FRA
			}
			any_sea_province = {
				limit = {
					has_province_modifier = loc_rename_modifier
				}
				secede_province = THIS
				
				state_scope = { change_region_name = "Renamer 2" }
				secede_province = NZL
			}
		}
		any_country = {
			clr_country_flag = rename_loc_region
		}
		any_sea_province = {
			remove_province_modifier = loc_rename_modifier
		}
		any_country = {
			limit = {
				tag = UTI
				check_variable = { which = itterator value = 0 }
			}
			change_variable = { which = itterator value = -1 }
			CLN = { country_event = 71602 }
		}
	}
	
	option = {
        name = "Change the Name"
    }
}
