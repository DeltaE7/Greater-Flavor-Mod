country_event = {
    id = 62027
    title = "The Peru-Bolivian dilemma!"
    desc = "62027_desc"
    picture = "deliberation"

    trigger = {
        exists = yes
		tag = GCO
		OR = {
			CHL = { has_country_flag = fight_the_PBC }
			CHL = { has_country_flag = peace_with_the_PBC }
			CHL = { war_with = PBC }
		}
        war = no
        NOT = {
            truce_with = PBC
            has_country_flag = fight_the_PBC
            has_country_flag = peace_with_the_PBC
            PBC = { has_country_flag = the_confederation_is_legitime }
        }
    }
    mean_time_to_happen = {
        months = 1
    }
    option = {
        name = "End the Peruvian threat once and for all!"
        set_country_flag = fight_the_PBC
        leave_alliance = PBC
		PEU = { all_core = { add_core = GCO }}
		SPU = { all_core = { add_core = GCO }}
        any_pop = {
            dominant_issue = {
                factor = 0.05
                value = jingoism
            }
        }
		GCO = { 
			war = {
				target = PBC
				attacker_goal = { casus_belli = cut_down_to_size_no_infamy country = GCO }
				defender_goal = { casus_belli = cut_down_to_size_no_infamy country = PBC }
				}
			war = {
				target = PEU
				attacker_goal = { casus_belli = cut_down_to_size_no_infamy country = GCO }
				defender_goal = { casus_belli = cut_down_to_size_no_infamy country = PEU }
				}
			war = {
				target = SPU
				attacker_goal = { casus_belli = cut_down_to_size_no_infamy country = GCO }
				defender_goal = { casus_belli = cut_down_to_size_no_infamy country = SPU }
				}
			}
        ai_chance = { factor = 1 }
    }
    option = {
        name = "We can't afford to intervene. Argentina and Chile will deal with them, surely."
        set_country_flag = peace_with_the_PBC
        prestige = -5
    GCO = { 
			relation = {
				who = PBC
				value = 100
        } 
	}
        any_pop = {
            dominant_issue = {
                factor = 0.05
                value = jingoism
            }
            scaled_militancy = {
                issue = jingoism
                factor = 4
				}
			}
        ai_chance = { factor = 0 }
		}
	}
