			duration = -1
		}
	}
}

country_event = {	
	id = 80003
	title = "MUH_PEOPLE"
	desc = "PIUPLES"
    is_triggered_only = yes

	option = {
		name = "MOOOOORE PEOPLEEEEE!"
        random_state = {
            limit = {
				any_owned_province = {
			        NOT = { has_province_modifier = america_new_life_2 }
					is_core = THIS
				}
            }
            any_owned = {
                limit = { is_core = THIS }
                add_province_modifier = { name = america_new_life_2 duration = -1 }
			}
            any_pop = {
                limit = {
                    pop_majority_ideology = conservative
                }
                militancy = 0.2
                consciousness = 0.4
            }
		}
	}
	
	option = {
	    name = "i want to try another state"
		country_event = 80004
	}
	
	option = {
	    name = "we have others things more importants to do"
	}
}

		name = "no more British claim on our land."

			limit = {
				AND = {

				}
			}

		}
	}
